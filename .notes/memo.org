* 清理分块日志文件

  #!/bin/bash
  KEEP_DAYS=28
  LOG_DIRS=(
      "/www/wwwroot/api/runtime/log"
  )

  for DIR in "${LOG_DIRS[@]}"; do
    if [ -d "$DIR" ]; then
      echo "正在清理目录: $DIR"
      find "$DIR" -type f -name "*.log" -mtime +$KEEP_DAYS -delete
      # 删除空的日期文件夹
      find "$DIR" -type d -empty -mindepth 1 -delete
    else
      echo "目录不存在: $DIR"
    fi
  done
  echo "✅ 日志清理完成，保留最近 $KEEP_DAYS 天内的日志。"
